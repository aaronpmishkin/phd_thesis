\begin{lemma}\label{lemma:convex-recursive-cone-constraint}
    The set \( \calCl \) is a convex cone for any \( l \geq 1 \).
\end{lemma}
\begin{proof}
    By construction, \( \calC^{(1)} = \R^{d_1 \times d_2} \)
    is a convex cone since any vector space is a convex cone.

    Assume \( \calCl \) is a convex cone for \( l \geq 1 \).
    If \( \Tll, H^{(l+1)} \in \calCll \), then
    for any \( \alpha, \beta \geq 0 \), \( \alpha \Tll + \beta H^{(l+1)} \)
    satisfies
    \begin{align*}
        \alpha \Tll_{j_l, i_{l+1}} + \beta H^{(l+1)}_{j_l, i_{l+1}}
         & = \alpha \Tl_{j_l \cdot i_{l+1}}
        - \alpha \Vl_{j_l \cdot i_{l+1}}
        + \beta H^{(l)}_{j_l \cdot i_{l+1}}
        - \beta U^{(l)}_{j_l \cdot i_{l+1}}                                           \\
         & = \rbr{\alpha \Tl_{j_l \cdot i_{l+1}} + \beta H^{(l)}_{j_l \cdot i_{l+1}}}
        - \rbr{\alpha \Vl_{j_l \cdot i_{l+1}} + \beta U^{(l)}_{j_l \cdot i_{l+1}}}
    \end{align*}
    where \( \alpha \Tl_{j_l \cdot i_{l+1}} + \beta H^{(l)}_{j_l \cdot i_{l+1}},
    \alpha \Vl_{j_l \cdot i_{l+1}} + \beta U^{(l)}_{j_l \cdot i_{l+1}}
    \in \calKl_{j_l} \) since \( \calKl_{j_l} \) is a convex cone.
    Similarly, \( \alpha \Tl + \beta H^{(l)} \in \calCl \)
    since \( \calCl \) is a convex cone by the inductive assumption.
    This implies \( \alpha \Tll + \beta H^{(l+1)} \in \calCll \)
    and we conclude that \( \calCll \) is a convex cone.
\end{proof}

\constraintConvexification*
\begin{proof}
    Recall that
    \( \calG^{(1)} = \R^{d_0 \times d_1} = \calC^{(1)} \);
    this establishes our base case.

    Now, assume that
    \( \calG^{(l)} = \calC^{(l)} \) for some \( 1 \leq l < k  \).
    Let \( \Tll \in \calG^{(l+1)} \) be arbitrary.
    By membership in \( \calG^{(l+1)} \), \( \Tll \) satisfies
    \[
        \Tll_{j_1, \ldots, j_l}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \Tl_{j_1, \ldots, j_{l-1}, i_l}
        \otimes W_{i_l}^{(l+1)},
    \]
    where \( \Tl_{i_l} \in \calKl_{j_l} \)
    and \( \Wll \in \R^{d_l \times d_{l+1}} \).
    For each \( i_{l+1} \in [d_{l+1}] \), this implies
    \begin{align*}
        \Tll_{j_l, i_{l+1}}
         & = \sum_{i_l \in \calI_{j_l}^{(l)}} \Tl_{i_l}
        \cdot W_{i_l, i_{l+1}}^{(l+1)}                       \\
         & = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0)
        - \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} < 0).
    \end{align*}
    We can now define
    \[
        H^{(l)}_{j_l \cdot i_{l+1}}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0)
    \]
    and
    \[
        V^{(l)}_{j_l \cdot i_{l+1}}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} < 0),
    \]
    to show that
    \( \Tll_{(j_l, i_{l+1})}
    = H^{(l)}_{j_l \cdot i_{l+1}} - V^{(l)}_{j_l \cdot i_{l+1}}\).
    Moreover,
    \( \calKl_{j_l} \) is a convex cone, so
    \( H^{(l)}_{j_l \cdot i_{l+1}}, V^{(l)}_{j_l \cdot i_{l+1}}
    \in \calKl_{j_l} \) as required.
    Finally, since \( \Tl \in \calG^{(l)} = \calC^{(l)} \), we know
    that \( H^{(l)} \) has a decomposition as follows:
    \begin{align*}
        H^{(l)}_{j_{l-1}, j_l \cdot i_{l+1}}
         & = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{j_{l-1}, i_l}
        \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0) \\
         & = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{
        \sbr{T^{(l-1)}_{j_{l-1} \cdot i_l} - T^{(l-1)}_{2j_{l-1} \cdot i_l}}
        \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0).
    \end{align*}
    Define \( \leftindex_{i_{l}}{T}^{(l)} \) be such that
    \[
        \leftindex_{i_{l}}{T}^{(l)}_{j_{l-1}, j_l \cdot i_{l+1}}
        =
        \sbr{T^{(l-1)}_{j_{l-1} \cdot i_l} - T^{(l-1)}_{2j_{l-1} \cdot i_l}}
        \mathds{1}(i_l \in \calI_{j_l}^{(l)}),
    \]
    which shows that
    \( \leftindex_{i_{l}}{T}^{(l)} \oplus \mathbf{0} \in \calCl \).
    But now we can write
    \[
        H^{(l)}_{j_{l-1}, j_l \cdot i_{l+1}}
        = \sum_{i_l \in [d_l]} \leftindex_{i_{l}}{T}^{(l)}
        \abs{W_{i_l, i_{l+1}}^{(l+1)}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0),
    \]
    which shows that \( H^{(l)} \oplus \mathbf{0} \in \calCl \) because \( \calCl \) is
    a convex cone (\cref{lemma:convex-recursive-cone-constraint}).
    Repeating this argument for \( \Vl \) shows that
    \( \Vl \oplus \mathbf{0} \in \calCl \) and thus
    \( H^{(l)} \oplus \Vl \oplus \mathbf{0} \in \calCl \).
    Note that these inclusions require \( d_{l} \geq 2 p_l d_{l+1} \),
    which holds by assumption.
    We conclude that \( \Tll \in \calCll \) and so
    \( \calG^{(l+1)} \subseteq \calC^{(l+1)} \).

    Now we show that the reverse inclusion is satisfied under
    the same assumption that \( d_{l} \geq 2 p_l d_{l+1} \).
    Towards that goal, take
    \( \Tll \in \calC^{(l+1)} \) to be arbitrary.
    By definition of \( \calC^{(l+1)} \), there exists
    \( \Tl \oplus \Vl \oplus \mathbf{0} \in \calC^{(l)} \) such that
    \begin{align*}
        \Tll_{j_l, i_{l+1}}
         & = \Tl_{j_l \cdot i_{l+1}} - \Vl_{j_l \cdot i_{l+1}},
    \end{align*}
    where \( \Tl_{j_l \cdot i_{l+1}}, \Vl_{j_l \cdot i_{l+1}}
    \in \calK_{j_l} \).
    That is,
    \[
        \Tll_{j_1, \ldots, j_l}
        = \sum_{i_{l+1}=1}^{d_{l+1}}
        \begin{bmatrix}
            \Tl_{j_1, \ldots, j_l \cdot i_{l+1}} \otimes e_{i_{l+1}}
            - \Vl_{j_1, \ldots, j_l \cdot i_{l+1}} \otimes e_{i_{l+1}}
        \end{bmatrix},
    \]
    where we are using standard basis vectors \( e_{i_{l+1}} \) to embed
    the tensors into the correct dimensions of \( \Tll \).
    Since \( C^{(l)} = G^{(l)} \),
    \( \Tl \oplus \Vl \oplus \mathbf{0} \in \calG^{(l)} \) as required.
    Choosing \( W^{(l+1)} \in \R^{d_{l} \times d_{l+1}} \) to be the block matrix
    with \( p_l + 1 \) blocks given by
    \[
        W^{(l+1)}
        =
        \begin{bmatrix}
            I      \\
            -I     \\
            \vdots \\
            I      \\
            -I     \\
            \mathbf{0}
        \end{bmatrix},
    \]
    we can now write
    \begin{align*}
        \Tll_{j_1, \ldots, j_{l}}
         & = \sum_{i_{l+1} = 1}^{d_{l+1}}
        \Tl_{j_1, \ldots, j_l \cdot i_{l+1}}
        \otimes W_{i_{l+1}}
        +
        \sum_{i_{l+1} = 1}^{d_{l+1}}
        \Vl_{j_1, \ldots, j_l \cdot i_{l+1}}
        \otimes W_{i_{l+1}}.
    \end{align*}
    Thus, we need \( 2 d_{l+1} \) outer-products per \( j_l \in [p_l] \) to
    represent \( \Tll \)
    This is possible since \( d_l \geq 2 p_l d_{l+1} \),
    by assumption.
    We conclude that \( \Tll \in \calGll \) and thus \( \calGll = \calCll \)
    as required.
\end{proof}
