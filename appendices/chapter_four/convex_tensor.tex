\begin{lemma}\label{lemma:convex-recursive-cone-constraint}
    The set \( \calCl \) is a convex cone for any \( l \geq 1 \).
\end{lemma}
\begin{proof}
    By construction, \( \calC^{(1)} = \R^{d_0 \times d_1} \)
    is a convex cone since every vector space is a convex cone.

    Assume \( \calCl \) is a convex cone for some \( l \geq 1 \).
    If \( \Tll, H^{(l+1)} \in \calCll \), then
    for any \( \alpha, \beta \geq 0 \),
    \( \alpha \Tll + \beta H^{(l+1)} \) satisfies
    \begin{align*}
        \alpha \Tll_{j_l, i_{l+1}} + \beta H^{(l+1)}_{j_l, i_{l+1}}
         & = \alpha \Tl_{ \phi_l(1, j_l, i_{l+1})}
        - \alpha \Tl_{ \phi_l(-1, j_l, i_{l+1})}
        + \beta H^{(l)}_{ \phi_l(1, j_l, i_{l+1})}
        - \beta H^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}                                                 \\
         & = \rbr{\alpha \Tl_{ \phi_l(1, j_l, i_{l+1})} + \beta H^{(l)}_{ \phi_l(1, j_l, i_{l+1})}}
        - \rbr{\alpha \Tl_{ \phi_l(-1, j_l, i_{l+1})} + \beta H^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}}
    \end{align*}
    where \( \alpha \Tl_{ \phi_l(1, j_l, i_{l+1})} + \beta H^{(l)}_{\phi_l(1, j_l, i_{l+1})},
    \alpha \Tl_{ \phi_l(-1, j_l, i_{l+1})} + \beta H^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}
    \in \calKl_{j_l} \) since \( \calKl_{j_l} \) is a convex cone.
    Similarly, \( \alpha \Tl + \beta H^{(l)} \in \calCl \)
    since \( \calCl \) is a convex cone by the inductive assumption.
    This implies \( \alpha \Tll + \beta H^{(l+1)} \in \calCll \)
    and we conclude that \( \calCll \) is a convex cone.
\end{proof}

\begin{lemma}\label{lemma:cone-scaling}
    If \( \leftindex_{k}{\Tl} \in \calCl \) and
    \( \leftindex_{k}{\alpha}_{i_{l}} \geq 0 \), then \( \hat{T}^{(l)} \) given
    by
    \begin{equation*}
        \hat{T}^{(l)}_{i_{l}}
        = \sum_{k=1}^{K}
        \leftindex_{k}{\alpha}_{i_{l}} \leftindex_{k}{\Tl}_{i_{l}}
    \end{equation*}
    also satisfies \( \hat{T}^{(l)} \in \calCl \).
\end{lemma}
\begin{proof}
    The proof follows by induction.
    If \( l = 1 \), then \( \calCl = \R^{d_0 \times d_1} \) and the result
    follows immediately.
    Now let \( l \geq 2 \) and suppose the result holds for \( l - 1 \).
    Since \( \leftindex_{k}{\Tl} \in \calCl \), it satisfies
    \[
        \leftindex_{k}{\Tl}_{j_{l-1}, i_l}
        = \leftindex_{k}{T}^{(l-1)}_{\phi_{l-1}(1, j_{l-1}, i_l)}
        - \leftindex_{k}{T}^{(l-1)}_{\phi_{l-1}(-1, j_{l-1}, i_l)}.
    \]
    Define \( \leftindex_{k}{\hat{T}}^{(l-1)} \) by
    \[
        \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(c, j_{l-1}, i_l)}
        = \leftindex_{k}{\alpha}_{i_{l}} \leftindex_{k}{T}^{(l-1)}_{\phi_{l-1}(c, j_{l-1}, i_l)}.
    \]
    The inductive assumption implies
    \( \leftindex_{k}{\hat{T}}^{(l-1)} \in \calC^{(l-1)} \).
    Moreover,
    \( \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(1, j_{l-1}, i_l)}
    \in \calK^{(l-1)}_{j_{l-1}} \) since \( \calK^{(l-1)}_{j_{l-1}} \) is a
    convex cone.
    Thus, the tensor \( \leftindex_{k}{\hat{T}}^{(l)} \) with entries
    \begin{equation*}
        \leftindex_{k}{\hat{T}}^{(l)}_{i_{l}} :=
        \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(1, j_{l-1}, i_l)}
        - \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(-1, j_{l-1}, i_l)}
    \end{equation*}
    must be in \( \calCl \).
    We conclude that \( \hat{T}^{(l)} \in \calCl \) since it is a sum (and
    therefore a non-negative combination ) of elements of \( \calCl \) and \(
    \calCl \) is a convex cone by
    \cref{lemma:convex-recursive-cone-constraint}.
\end{proof}

\constraintConvexification*
\begin{proof}
    Recall that
    \( \calG^{(1)} = \R^{d_0 \times d_1} = \calC^{(1)} \);
    this establishes our base case.

    Now, assume that
    \( \calG^{(l)} = \calC^{(l)} \) for some \( 1 \leq l < L  \).
    Let \( \Tll \in \calG^{(l+1)} \) be arbitrary.
    By membership in \( \calG^{(l+1)} \), \( \Tll \) satisfies
    \[
        \Tll_{j_1, \ldots, j_l}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \Tl_{j_1, \ldots, j_{l-1}, i_l}
        \otimes W_{i_l}^{(l+1)},
    \]
    where \( \Tl_{i_l} \in \calKl_{j_l} \)
    and \( \Wll \in \R^{d_l \times d_{l+1}} \).
    For each \( i_{l+1} \in [d_{l+1}] \), this implies
    \begin{align*}
        \Tll_{j_l, i_{l+1}}
         & = \sum_{i_l \in \calI_{j_l}^{(l)}} \Tl_{i_l}
        \cdot W_{i_l, i_{l+1}}^{(l+1)}                       \\
         & = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0)
        - \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} < 0).
    \end{align*}
    We can now define
    \[
        \hat{T}^{(l)}_{ \phi_l(1, j_l, i_{l+1})}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0)
    \]
    and
    \[
        \hat{T}^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} < 0),
    \]
    to show that
    \( \Tll_{j_l, i_{l+1}}
    = \hat{T}^{(l)}_{ \phi_l(1, j_l, i_{l+1})} - \hat{T}^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}\).
    Because
    \( \calKl_{j_l} \) is a convex cone,
    \( \hat{T}^{(l)}_{ \phi_l(1, j_l, i_{l+1})}, \hat{T}^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}
    \in \calKl_{j_l} \) as required.
    Finally, since \( \Tl \in \calG^{(l)} = \calC^{(l)} \), we can invoke
    \cref{lemma:cone-scaling} to show that \( \hat{T}^{(l)} \in \calC \).
    Note that these inclusions require \( d_{l} \geq 2 p_l d_{l+1} \),
    which holds by assumption.
    We conclude that \( \Tll \in \calCll \) and so
    \( \calG^{(l+1)} \subseteq \calC^{(l+1)} \).

    Now we show that the reverse inclusion is satisfied under
    the same assumption that \( d_{l} \geq 2 p_l d_{l+1} \).
    Towards that goal, take
    \( \Tll \in \calC^{(l+1)} \) to be arbitrary.
    By definition of \( \calC^{(l+1)} \), there exists
    \( \Tl \in \calCl \) such that
    \begin{align*}
        \Tll_{j_l, i_{l+1}}
         & = \Tl_{ \phi_l(1, j_l, i_{l+1})} - \Tl_{ \phi_l(-1, j_l, i_{l+1})},
    \end{align*}
    where \( \Tl_{ \phi_l(1, j_l, i_{l+1})}, \Tl_{ \phi_l(-1, j_l, i_{l+1})}
    \in \calK_{j_l} \).
    That is,
    \[
        \Tll_{j_1, \ldots, j_l}
        = \sum_{i_{l+1}=1}^{d_{l+1}}
        \begin{bmatrix}
            \Tl_{j_1, \ldots,  \phi_l(1, j_l, i_{l+1})} \otimes e_{i_{l+1}}
            - \Tl_{j_1, \ldots,  \phi_l(-1 j_l, i_{l+1})} \otimes e_{i_{l+1}}
        \end{bmatrix},
    \]
    where we are using standard basis vectors \( e_{i_{l+1}} \) to embed
    the tensors into the correct dimensions of \( \Tll \).
    Since \( C^{(l)} = G^{(l)} \),
    \( \Tl \in \calG^{(l)} \) as required.
    Choosing \( W^{(l+1)} \in \R^{d_{l} \times d_{l+1}} \) to be the block matrix
    with \( p_l + 1 \) blocks given by
    \[
        W^{(l+1)}
        =
        \begin{bmatrix}
            I      \\
            \vdots \\
            I      \\
            -I     \\
            \vdots
            -I     \\
            \mathbf{0}
        \end{bmatrix},
    \]
    we can now write
    \begin{align*}
        \Tll_{j_1, \ldots, j_{l}}
         & = \sum_{i_{l} \in \calI_{j_l}^{(l)}}
        \Tl_{j_1, \ldots, j_{l-1}, i_l}
        \otimes W_{i_{l}},
    \end{align*}
    where the activation indices are given by
    \[
        \calI_{j_l}^{(l)}
        = \cbr{ \phi_l(1, j_l, i_{l+1}) : i_{l+1} \in [d_{l+1}]}
        \cup
        \cbr{ \phi_l(-1, j_l, i_{l+1}) : i_{l+1} \in [d_{l+1}]}.
    \]
    Thus, we need \( 2 d_{l+1} \) outer-products per \( j_l \in [p_l] \) to
    represent \( \Tll \).
    This is possible since \( d_l \geq 2 p_l d_{l+1} \),
    by assumption.
    We conclude that \( \Tll \in \calGll \) and thus \( \calGll = \calCll \)
    as required.
\end{proof}

\tensorSolutionMapping*
\begin{proof}
    The proof reverses the layer elimination lemma.
    For any \( l \in [k-1] \), we have
    \begin{align*}
        X^{(l+1)} \odot T^{(l+1)}
         & = \sum_{j_{l}=1}^{p_l} X^{(l+1)}_{j_{l}} \odot T^{(l+1)}_{j_{l}}                \\
         & = \sum_{j_{l}=1}^{p_l} \sum_{i_{l+1}=1}^{d_{l+1}}
        \rbr{X^{(l+1)}_{j_{l}} \odot T^{(l+1)}_{j_{l}, i_{l+1}}} \otimes e_{i_{l+1}}       \\
         & = \sum_{j_{l}=1}^{p_l} \sum_{i_{l+1}=1}^{d_{l+1}}
        \rbr{X^{(l+1)}_{j_{l}} \odot
        \sbr{T^{(l)}_{\phi_l(1, j_{l}, i_{l+1})} - T^{(l)}_{\phi_l(-1, j_{l}, t i_{l+1})}}}
        \otimes e_{i_{l+1}}                                                                \\
         & = \sum_{j_{l}=1}^{p_l} \sum_{i_{l+1}=1}^{d_{l+1}} \sum_{s \in \cbr{+1, -1}}
        \rbr{X^{(l+1)}_{j_{l}} \odot
        T^{(l)}_{\phi_l(s, j_{l}, i_{l+1})}} \otimes W^{(l+1)}_{\phi_l(s, j_{l}, i_{l+1})} \\
         & = \sum_{i_{l}=1}^{d_{l}}
        D^{(l)}_{i_{l}} \rbr{X^{(l)} \odot
        T^{(l)}_{i_{l}}} \otimes W^{(l+1)}_{i_{l}}                                         \\
         & = \rbr{X^{(l)} \odot T^{(l)}}_+ W^{(l+1)}.
    \end{align*}
    Applying this recursively from \( l = k - 1 \)
    and using \( X^{(1)} \odot T^{(1)} = X W^{(1)} \) completes the proof.
\end{proof}

\minimumSolutionWidth*
\begin{proof}
    Let \( \Tk \in \calCk \) be a solution to
    Problem~\ref{eq:convex-tensor-program};
    we will construct a solution with the desired width bounds using
    \( \Tk \).
    Towards this goal, let \( l \in [L-1] \) and suppose that the layer at
    \( l+1 \) has \( k \) non-zero outputs.
    That is, the set
    \[
        \calA(\Tll) = \cbr{i_{l+1} \in [d_{l+1}] : T^{(l+1)}_{i_{l+1}} \neq 0 },
    \]
    has \( k \) elements.
    Now, observe that for each \( i_{l+1} \in \calA(T^{(l+1)}) \),
    \begin{align*}
        X^{(l+1)} \odot T^{(l+1)}_{i_{l+1}}
         & = \sum_{j_{l} = 1}^{p_{l}}
        X^{(l+1)}_{j_l}  \odot T^{(l+1)}_{j_{l}, i_{l+1}} \\
         & = \sum_{j_{l} = 1}^{p_{l}}
        X^{(l+1)}_{j_l}  \odot
        \sbr{T^{(l)}_{\phi_{l}(1, j_{l}, i_{l+1})} - T^{(l)}_{\phi_{l}(-1, j_{l}, i_{l+1})}}.
    \end{align*}
    The non-zero terms in this sum must each be embedded into \( T^{(l)} \),
    meaning each contributes one to the width of the
    \( (l+1)^{\text{th}} \) layer.
    The set of non-zero terms is,
    \[
        \calQ_{i_{l+1}}(T^{(l)})
        = \cbr{q_{s, j_l, i_{l+1}} :=
            X^{(l+1)}_{j_l} \odot T^{(l)}_{\phi_{l}(s, j_{l}, i_{l+1})}
            : j_l \in [p_l], s \in \cbr{+1, -1},
            T^{(l)}_{\phi_{l}(s, j_{l}, i_{l+1})} \neq 0}
    \]
    Since each vector \( q_{s, j_l, i_{l+1}} \) is \( n \)-dimensional,
    \( \calQ_{i_{l+1}}(\Tl) \) contains a linearly dependent set if it contains
    more than \( n \) vectors.
    Let
    \[
        \calN_{i_{l+1}}(T^{(l)})
        = \cbr{(s, j_l) : q_{s, j_l, i_{l+1}} \in \calQ_{i_{l+1}}(\Tl)}.
    \]
    Linear dependence implies there exists
    \( \cbr{\alpha_{s,j_l} : (s,j_l) \in \calN_{i_{l+1}}(T^{(l)})} \)
    not all equal to zero such that
    \begin{align*}
        0 & = \sum_{(s,j_l) \in \calN_{i_{l+1}}(T^{(l)})} \alpha_{s, j_l} q_{s, j_l, i_{l+1}} \\
          & = \sum_{(s,j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        \alpha_{s,j_l} \rbr{X^{(l)}_{j_l} \odot T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}}          \\
          & = \sum_{(s,j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        \rbr{X^{(l+1)}_{j_l} \odot \sbr{\alpha_{s,j_l} T^{(l)}_{\phi_{l}(s,j_l,i_{l+1})}}}.
    \end{align*}
    Let \( (s^*, r^*) =
    \argmax_{s, j_l} \cbr{ \abs{\alpha_{s, j_l}} :
        (s, j_l) \in \calN_{i_{l+1}}(T^{(l)})} \)
    and define \( \tilde{\alpha}_{s, j_l} = \alpha_{s, j_l} / \abs{\alpha_{s^* ,r^*}} \).
    Clearly \( \abs{\tilde{\alpha}_{s, j_l}} \leq 1 \) for all \( (s, j_l) \)
    and \( \abs{\tilde{\alpha}_{(s^*, r^*)}} = 1 \).
    Using this notation, we have
    \begin{align*}
        0 & = \sum_{(s, j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        \rbr{X^{(l)}_{j_l} \odot \sbr{\tilde{\alpha}_{s, j_l} T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}}}.
    \end{align*}
    This implies
    \begin{align*}
        X^{(l+1)} \odot T^{(l+1)}_{i_{l+1}}
         & = \sum_{(s, j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        X^{(l+1)}_{j_l} \odot
        \sbr{(1 + \tilde{\alpha}_{s, j_l}) T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}} \\
         & = \sum_{(s,j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        X^{(l+1)}_{j_l} \odot
        \hat{T}^{(l)}_{\phi_{l}(s, j_l, i_{l+1})},
    \end{align*}
    where \( \hat{T^{(l)}} \) is the scaled version of \( T^{(l)} \)
    satisfying
    \( \hat{T}^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}
    = \sbr{(1 + \tilde{\alpha}_{s, j_l}) T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}} \)
    for each \( (s, j_l) \in \calN_{i_{l+1}}(T^{(l)}) \) and
    \( \hat{T}^{(l)}_{\phi_{l}(s, j_l, i_{l+1})} = T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})} \)
    otherwise.
    It is clear from the above calculation that \( \hat{T}^{(l)} \) preserves
    the predictions of \( T^{(l)} \)
    and maintains \( \hat{T}^{(l)}_{\phi_{l}(s, j_l, i_{l+1})} \in \calK_{j_l}^{(l)} \)
    since \( \abs{\tilde{\alpha}_{s, j_l}} \leq 1 \) for each \( (s, j_l) \).
    Additionally, \( \hat{T}^{(l)} \in \calCl \) by \cref{lemma:cone-scaling}.
    Defining
    \[
        \hat{T}^{(l+1)}_{s, j_l, i_{l+1}} =
        \hat{T}^{(l)}_{\phi_{l}(1, j_l, i_{l+1})}
        - \hat{T}^{(l)}_{\phi_{l}(-1, j_l, i_{l+1})},
    \]
    we see that \( \hat{T}^{(l+1)} \in \calC^{(l+1)} \)
    and thus \( \hat{T}^{(l+1)} \) is also optimal.
    Moreover,
    \( \calA(\hat{T}^{(l)}) \leq \calA(\hat{T}^{(l)}) - 1 \).
    In other words, the width of \( m_l(\Tk) \) is reduced by one.

    For each \( i_{l+1} \), we can repeat this argument until the vectors
    in \( \calQ_{i_{l+1}}(\hat{T}^{(l)}) \) are linearly independent.
    This implies there exists optimal \( \hat{T}^{(l)} \)
    for which \( \abs{\calQ_{i_{l+1}}(\hat{T}^{(l)})} \leq n \).
    Since this procedure is performed independently for each \( i_{l+1} \)
    for which \( T^{(l+1)}_{i_{l+1}} \neq 0 \), we construct
    \( \hat{T}^{(l+1)} \) with
    \( m_{l}(\hat{T}^{(l)}) \leq k n \).
    Noting that \( k \leq c \) (the number of network outputs) when
    \( l = k - 1 \) and applying a simple inductive argument completes the
    proof.
\end{proof}

\deepConvexReformulation*
\begin{proof}
    Let \( \Tk \in \calCk \) be the sparse solution to
    Problem~\ref{eq:convex-tensor-program}
    which is guaranteed to exist by \cref{prop:convex-minimal-solution-width}.
    This solution satisfies \( m_l = \calA\rbr{\Tl} \leq c n^{L - l} \).
    \cref{lemma:convex-tensor-solution-mapping} now implies that there
    exists an optimal sparse neural network \( \theta^* \) with at
    most \( m_l \) non-zero neurons in the \( l^{\text{th}} \) layer.
    Removing zero neurons results in the architecture
    \( \Delta' = \rbr{m_1, \ldots, m_{L-1}} \)
    with associated parameter space \( \Theta_{\Delta'} \).
    Let \( \Delta \) be any architecture satisfying
    \( d_l \geq 2 p_l d_{l+1} \).
    Now can obtain
    \begin{align*}
        \min_{\Tk \in \calCk} \calL(\Xk \odot \Tk, y)
         & = \min_{\theta \in \Theta_{\Delta}} \calL(f_\theta(X), y)     \\
         & \leq \min_{\theta \in \Theta_{\Delta'}} \calL(f_\theta(X), y) \\
         & \leq \calL(f_{\theta^*}(X), y)                                \\
         & = \min_{\Tk \in \calCk} \calL(\Xk \odot \Tk, y),
    \end{align*}
    which shows that the two problems have the same optimal value.
\end{proof}

\gatedReLULayerElimination*
\begin{proof}
    We start by defining a suitable \( \Tll \) as follows:
    \begin{equation}\label{eq:gated-t-plus-characterization}
        \Tll_{j_1, \ldots, j_l}
        = \Tl_{j_1, \ldots, j_{l-1}, j_l} \otimes W^{(l+1)}_{j_l}.
    \end{equation}
    Using this notation, we re-write the sum
    in \cref{eq:gated-layer-to-tensorize} as follows:
    \begin{align*}
        Z^{(l+2)}
         & = \mathbf{D}^{(l+1)}_\text{gated} \circ
        \sbr{\sum_{j_l=1}^{p_l^{\text{gated}}} \rbr{\Dl_{j_l} \sbr{\tXl \odot \Tl_{j_l}}}
        \otimes W^{(l+1)}_{j_l}}                                                      \\
         & = \mathbf{D}^{(l+1)}_\text{gated} \circ
        \sbr{\sum_{j_l=1}^{p_l^{\text{gated}}} \rbr{\Dl_{j_l} \tXl \odot \sbr{\Tl_{j_l}
        \otimes W^{(l+1)}_{j_l}}}}                                                    \\
         & = \mathbf{D}^{(l+1)}_\text{gated} \circ
        \sbr{\sum_{j_l=1}^{p_l^{\text{gated}}} \rbr{\Dl_{j_l} \tXl \odot \Tll_{j_l}}} \\
         & = \mathbf{D}^{(l+1)}_\text{gated} \circ \sbr{\tXll \odot \Tll}.
    \end{align*}
    It should be clear from the way that we constructed \( \Tll \) that
    it is contained \( \calH^{(l+1)} \).
    This establishes one direction of theorem.
    The other direction can be shown by instead assuming that
    \( \Tll \in \calH^{(l+1)} \) and then noting that the characterization
    in \cref{eq:gated-t-plus-characterization} immediately follows for some
    \( \Tl \in \calH^{(l)} \) and \( W^{(l+1)} \in \R^{p_l \times p_{l+1}} \).
    The preceding calculation is then reversed to obtain
    \cref{eq:gated-layer-to-tensorize}.
\end{proof}

\convexGatedReLU*
\begin{proof}
    Let \( \theta = (W^{(1)}, \ldots, W^{(L)}) \) be any optimal
    solution to \cref{eq:duplicated-gated-relu-program}.
    We will define a corresponding tensor model with the
    same predictions.
    Let \( T^{(1)} = W^{(1)} \).
    This choice trivially satisfies
    \begin{align*}
        g_{\theta_1}^{\text{dup}}(X)
         & = \mathbf{D}^{(1)}_{\text{dup}} \circ X W^{(1)}          \\
         & = \mathbf{D}^{(1)}_{\text{dup}} \circ (X \odot T^{(1)}).
    \end{align*}
    Now assume that for \( l \in [L-1] \),
    \( g_{\theta_l}^{\text{dup}}(X) = \mathbf{D}^{(1)}_{\text{dup}} \circ (X \odot T^{(l)}) \).
    We will show that this equation also holds at \( l+1 \).
    Let \( \calI^{(l)}_{j_l} \) contain the indices of the parameters in the
    layer \( l \) which are associated with the \( d_{l+1} \) copies of
    \( D^{(l)}_{j_l} \).
    Let
    \begin{equation*}
        \Tll_{j_1, \ldots, j_l}
        = \sum_{i_l \in \calI^{(l)}_{j_l}} \Tl_{j_1, \ldots, j_{l-1}, i_l} \otimes W^{(l+1)}_{i_l}.
    \end{equation*}
    Using this notation, we re-write the feature map
    for the duplicated gated ReLU model as follows:
    \begin{align*}
        g_{\theta_{l+1}}^{\text{dup}}(X)
         & = \mathbf{D}^{(l+1)}_\text{dup} \circ
        \sbr{\sum_{i_l=1}^{d_l}
            \rbr{\Dl_{i_l} \sbr{\tXl \odot \Tl_{i_l}}}
        \otimes W^{(l+1)}_{i_l}}                                                      \\
         & = \mathbf{D}^{(l+1)}_\text{dup} \circ
        \sbr{\sum_{j_l=1}^{p_l^{\text{gated}}} \sum_{i_l \in \calI^{(l)}_{j_l}}
        \rbr{\Dl_{j_l} \sbr{\tXl \odot \Tl_{i_l}}}
        \otimes W^{(l+1)}_{i_l}}                                                      \\
         & = \mathbf{D}^{(l+1)}_\text{dup} \circ
        \sbr{\sum_{j_l=1}^{p_l^{\text{gated}}} \rbr{\Dl_{j_l} \tXl \odot \sbr{\sum_{i_l \in \calI^{(l)}_{j_l}} \Tl_{i_l}
        \otimes W^{(l+1)}_{i_l}}}}                                                    \\
         & = \mathbf{D}^{(l+1)}_\text{dup} \circ
        \sbr{\sum_{j_l=1}^{p_l^{\text{gated}}} \rbr{\Dl_{j_l} \tXl \odot \Tll_{j_l}}} \\
         & = \mathbf{D}^{(l+1)}_\text{dup} \circ \sbr{\tXll \odot \Tll}.
    \end{align*}
    Induction (with a minor modification to drop the activation patterns
    from the final layer) now implies that
    \[
        g_{\theta}^{\text{dup}}(X) = \tXk \odot \Tk.
    \]
    This implies that
    \[
        \min_{\theta = (W^{(1)}, \ldots, \Wk)} \calL\rbr{g^{\text{dup}}_{\theta}(X), y}
        \geq
        \min_{\Tk} \calL\rbr{\tXk \odot \Tk, y}.
    \]
    This establishes one direction of theorem.

    For the other direction, let \( \Tk \) be any optimal solution
    to \cref{eq:convex-gated-relu-program}.
    We shall construct a parameter \( \theta = \rbr{W^{(1)}, \ldots \Wk} \)
    for the duplicated Gated ReLU model with the same model fit.
    We have
    \begin{align*}
        \tXll \odot \Tll
         & = \sum_{j_l = 1}^{p_l^{\text{gated}}}
        \Dl_{i_l} \rbr{\tXl \odot \Tll_{j_l}}                                     \\
         & = \sum_{j_l = 1}^{p_l^{\text{gated}}}
        \Dl_{j_l} \rbr{\tXl \odot
        \sbr{\sum_{i_{l+1}=1}^{d_{l+1}} \Tll_{j_l, i_{l+1}} \otimes e_{i_{l+1}}}} \\
         & = \sum_{j_l = 1}^{p_l^{\text{gated}}}
        \sum_{i_{l+1}=1}^{d_{l+1}} \sbr{\Dl_{j_l} \rbr{\tXl \odot
        \Tll_{j_l, i_{l+1}}}} \otimes e_{i_{l+1}}.                                \\
        \intertext{Define \( \Tl_{\phi_l(j_l, i_{l+1})} = \Tll_{j_l, i_{l+1}} \),
            which is possible since \( d_l \geq p_l d_{l+1} \)
            Similarly, let \( \Wll_{\phi(j_l, i_{l+1})} = e_{i_{l+1}} \) to obtain
        }
         & =\sum_{i_l}^{d_{l+1}} \sbr{\Dl_{i_l} \rbr{\tXl \odot
        \Tl_{i_l}}} \otimes \Wll_{i_{l}}                                          \\
         & =\mathbf{D}^{(l)}_{\text{dup}} \circ \rbr{\tXl \odot
            \Tl} \Wll
    \end{align*}
    Unrolling this equation starting from \( l = L-1 \) and choosing
    \( W^{(1)} = T^{(1)} \) yields  \( \theta = \rbr{W^(1), \ldots, W^{(L)}} \)
    such that
    \[
        \tXk \odot \Tk
        = g_{\theta}^{\text{dup}}(X),
    \]
    as desired.
    We conclude that
    \[
        \min_{\Tk} \calL\rbr{\tXk \odot \Tk, y}
        \geq
        \min_{\theta = (W^{(1)}, \ldots, \Wk)} \calL\rbr{g^{\text{dup}}_{\theta}(X), y},
    \]
    which completes the proof.
\end{proof}

\begin{restatable}[Duplicated Gated ReLU Layer Elimination]{proposition}{duplicatedLayerElimination}%
    \label{prop:duplicated-layer-elimination}
    If there exists
    \( H^{(l)}, \Tl \in
    \R^{d_0 \times p_1^{\text{gated}} \times \cdots \times p_{l-1}^{\text{gated}} \times d_{l}} \)
    such that the gated ReLU model at layer \( l+1 \) satisfies
    \begin{equation}\label{eq:duplicated-layer-to-tensorize}
        \Zll \Wll =
        \sbr{\mathds{1}(\tXl \odot H^{(l)} \geq 0) \circ
            (\tXl \odot \Tl)} \Wll
    \end{equation}
    then there exists
    \( \Tll \in \R^{d_0 \times p_1^{\text{gated}} \times \cdots \times p_{l}^{\text{gated}} \times d_{l+1}} \)
    satisfying
    \[
        \Zll \Wll = \tXll \odot \Tll
    \]
    and
    \( H^{(l+1)} \in \R^{d_0 \times p_1^{\text{gated}} \times \cdots \times p_{l}^{\text{gated}} \times d_{l+1}} \)
    such that
    \[
        Z^{(l+2)} W^{(l+2)}
        = \sbr{\mathds{1}(\tXll \odot H^{(l+1)} \geq 0) \circ \rbr{\tXll \odot \Tll}} W^{(l+2)}.
    \]
\end{restatable}
\begin{proof}
    Let \( \calI^{(l)}_{j_l} \) be the indices of gate vectors mapping to the
    \( j_l^{\text{th}} \) pattern,
    \[
        \calI^{(l)}_{j_l}
        := \cbr{ i_l \in [d_l] : \Dl_{j_l} = \diag\rbr{\mathds{1}(\tXl \odot H^{(l)}_{i_l} \geq 0)}}.
    \]
    Define a candidate tensor as
    \[
        \Tll_{j_1, \ldots, j_{l}}
        = \sum_{i_l \in \calI^{(l)}_{j_l}} \Tl_{j_1, \ldots, j_{l-1}, i_l} \otimes \Wll_{i_{l}}
    \]
    Now we can write the pre-activations at layer \( l+1 \) as follows:
    \begin{align*}
        \Zll \Wll
         & = \sbr{\mathds{1}(\tXl \odot H^{(l)} \geq 0) \circ (\tXl \odot \Tl)} \Wll                   \\
         & = \sum_{i_l=1}^{d_l}
        \sbr{\diag\rbr{\mathds{1}(\tXl \odot H_{i_l}^{(l)} \geq 0)} (\tXl \odot \Tl_{i_l})} \Wll_{i_l} \\
         & = \sum_{j_l=1}^{p^{\text{gated}}_l}
        \sum_{i_l \in \calI^{(l)}_{j_l}}
        \sbr{\Dl_{j_l} (\tXl \odot \Tl_{i_l})} \Wll_{i_l}                                              \\
         & = \sum_{j_1=1}^{p^{\text{gated}}_1} \cdots \sum_{j_l=1}^{p^{\text{gated}}_l}
        \Dl_{j_l} \sbr{\tXl}_{j_1, \ldots, j_{l-1}}
        \rbr{\sum_{i_l \in \calI^{(l)}_{j_l}} \Tl_{j_1, \ldots, j_{l-1}, i_l} \otimes \Wll_{i_l}}      \\
         & = \sum_{j_1=1}^{p^{\text{gated}}_1} \cdots \sum_{j_l=1}^{p^{\text{gated}}_l}
        \Dl_{j_l} \sbr{\tXl}_{j_1, \ldots, j_{l-1}} \Tll_{j_1, \ldots, j_{l-1}, j_l}                   \\
         & =  \tXll \odot \Tll.
    \end{align*}
    This completes the first part of the proof.
    Moreover, substituting \( G^{(l+1)} \) for \( \Wll \) in the above
    argument shows that there exists some \( H^{(l+1)} \) satisfying
    \[
        \Zll G^{(l+1)} = \tXll \odot H^{(l+1)}.
    \]
    As a result,
    \begin{align*}
        Z^{(l+2)} W^{(l+2)}
         & = \sbr{\mathds{1}(\Zll G^{(l+1)} \geq 0) \circ \rbr{\tXll \odot \Tll}} W^{(l+2)}         \\
         & = \sbr{\mathds{1}(\tXll \odot H^{(l+1)} \geq 0) \circ \rbr{\tXll \odot \Tll}} W^{(l+2)},
    \end{align*}
    as desired.
\end{proof}

\duplicatedEquivalence*
\begin{proof}
    Let \( \theta = \rbr{W^{(1)}, \ldots, W^{(L)}} \)
    be the parameters of a gated ReLU model and \( G^{(1)}, \ldots, G^{(L)} \)
    the associated gates.
    Let \( T^{(1)} = W^{(1)} \).
    Induction applied to \cref{prop:duplicated-layer-elimination} implies
    that there exists
    \( \Tk \in \R^{d_0 \times p_1^\text{gated} \times \ldots p_{L-1}^{\text{gated}} \times d_L} \)
    satisfying
    \[
        g_\theta(X) = \Xk \odot \Tk.
    \]
    \cref{thm:convex-gated-relu} and the proof thereof shows that every
    duplicated gated ReLU model is equivalent to an unconstrained tensor
    \( \Tk \) and vice versa.
    As a result, we have shown that every standard gated ReLU model can be
    represented as a duplicated gated ReLU model.

    Unfortunately, the converse does not hold.
    To see this, consider a duplicated gated ReLU model where the intermediate
    feature map \( g_{\theta_{l}}^{\text{dup}} \) is constrained to lie in a
    subspace (this can be achieved by having only a small number of non-zero
    neurons).
    This feature map will not be able to achieve all of the activation patterns
    in the duplicated gated ReLU model at layer \( l+1 \),
    meaning it cannot be written as a gated ReLU network.
\end{proof}

\subsection{Deep Cone Decompositions: Proofs}\label{app:deep-cone-decompositions}

\deepNonSingularCones*
\begin{proof}
    The proof is straightforward.
    By construction, \( \calC_{\text{ns}}^{(1)} = \R^{d_0 \times d_1} \).
    Assume that
    \( \calCl_{\text{ns}} = \R^{d_0 \times p^\text{ns}_1 \cdots \times p^\text{ns}_{l-1} \times d_{l}} \)
    for some \( l \in [L-1] \).
    Consider any \( \Tll \in \R^{d_0 \times p^\text{ns}_1 \cdots \times p^\text{ns}_{l} \times d_{l+1}} \).
    Since \( \calK_{j_l} \) is non-singular, there exists
    \( \Tl_{\phi_l(1, j_l, i_{l+1})}, \Tl_{ \phi_l(-1, j_l, i_{l+1})}
    \in \calK^{(l)}_{j_{l}} \) such that
    \[
        \Tll_{j_{l}, i_{l+1}}
        = \Tl_{ \phi_l(1, j_l, i_{l+1})}
        - \Tl_{ \phi_l(-1, j_l, i_{l+1})}, \\
    \]
    Moreover,
    \( \Tl \in \calCl_{\text{ns}} \) trivially by the inductive assumption.
    But this implies that \( \Tll \in \calCll_{\text{ns}} \) and thus
    \( \calCll_{\text{ns}} \supseteq \R^{d_0 \times p^\text{ns}_1 \cdots \times p^\text{ns}_{l-1} \times d_{l}} \).
    This completes the proof.
\end{proof}

\deepConeElimination*
\begin{proof}
    The proof is a straightforward application of \cref{thm:cone-elimination}.
    Indeed, \cref{thm:cone-elimination} shows that the space of two-layer
    convex ReLU networks is not affect by removing singular
    cones.
    That is, every two-layer convex ReLU model \( T^{(2)} \in \calC^{(2)} \)
    satisfies
    \[
        X^{(2)} \odot T^{(2)} = X_{\text{ns}}^{(2)} \odot T^{(2)'}
    \]
    for some model \( T^{(2)'} \in \calC^{(2)}_{\text{ns}} = \R^{d_0 \times p_1 \times d_2} \).

    Since \( X_{\text{ns}}^{(2)} \odot T^{(2)'} \) is a linear function of two arguments,
    we can represent both \( X^{(2)}_{\text{ns}} \) and \( T^{(2)'} \)
    as matrices.
    In particular, we can write the problem of training a two-layer
    ReLU network on the matrix version of \( X^{(2)}_{\text{ns}} \).
    Note that \cref{prop:layer-elimination} shows that this is equivalent
    to training a three-layer ReLU network.
    Applying \cref{thm:cone-elimination} again and repeating the argument
    above is sufficient to show that we may omit all singular cones from
    the second hidden layer.
    That is, we may use \( X^{(3)}_{\text{ns}} \) in place of \( X^{(3)} \).
    Arguing by induction in this fashion is sufficient to complete the proof.
\end{proof}

\coneDecompFormulation*
\begin{proof}
    The result follows immediately from \cref{prop:deep-cone-elimination}
    and \cref{thm:deep-convex-reformulation}.
\end{proof}
