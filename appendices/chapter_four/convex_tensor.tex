\begin{lemma}\label{lemma:convex-recursive-cone-constraint}
    The set \( \calCl \) is a convex cone for any \( l \geq 1 \).
\end{lemma}
\begin{proof}
    By construction, \( \calC^{(1)} = \R^{d_0 \times d_1} \)
    is a convex cone since every vector space is a convex cone.

    Assume \( \calCl \) is a convex cone for some \( l \geq 1 \).
    If \( \Tll, H^{(l+1)} \in \calCll \), then
    for any \( \alpha, \beta \geq 0 \),
    \( \alpha \Tll + \beta H^{(l+1)} \) satisfies
    \begin{align*}
        \alpha \Tll_{j_l, i_{l+1}} + \beta H^{(l+1)}_{j_l, i_{l+1}}
         & = \alpha \Tl_{ \phi_l(1, j_l, i_{l+1})}
        - \alpha \Tl_{ \phi_l(-1, j_l, i_{l+1})}
        + \beta H^{(l)}_{ \phi_l(1, j_l, i_{l+1})}
        - \beta H^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}                                                 \\
         & = \rbr{\alpha \Tl_{ \phi_l(1, j_l, i_{l+1})} + \beta H^{(l)}_{ \phi_l(1, j_l, i_{l+1})}}
        - \rbr{\alpha \Tl_{ \phi_l(-1, j_l, i_{l+1})} + \beta H^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}}
    \end{align*}
    where \( \alpha \Tl_{ \phi_l(1, j_l, i_{l+1})} + \beta H^{(l)}_{\phi_l(1, j_l, i_{l+1})},
    \alpha \Tl_{ \phi_l(-1, j_l, i_{l+1})} + \beta H^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}
    \in \calKl_{j_l} \) since \( \calKl_{j_l} \) is a convex cone.
    Similarly, \( \alpha \Tl + \beta H^{(l)} \in \calCl \)
    since \( \calCl \) is a convex cone by the inductive assumption.
    This implies \( \alpha \Tll + \beta H^{(l+1)} \in \calCll \)
    and we conclude that \( \calCll \) is a convex cone.
\end{proof}

\begin{lemma}\label{lemma:cone-scaling}
    If \( \leftindex_{k}{\Tl} \in \calCl \) and
    \( \leftindex_{k}{\alpha}_{i_{l}} \geq 0 \), then \( \hat{T}^{(l)} \) given
    by
    \begin{equation*}
        \hat{T}^{(l)}_{i_{l}}
        = \sum_{k=1}^{K}
        \leftindex_{k}{\alpha}_{i_{l}} \leftindex_{k}{\Tl}_{i_{l}}
    \end{equation*}
    also satisfies \( \hat{T}^{(l)} \in \calCl \).
\end{lemma}
\begin{proof}
    The proof follows by induction.
    If \( l = 1 \), then \( \calCl = \R^{d_0 \times d_1} \) and the result
    follows immediately.
    Now let \( l \geq 2 \) and suppose the result holds for \( l - 1 \).
    Since \( \leftindex_{k}{\Tl} \in \calCl \), it satisfies
    \[
        \leftindex_{k}{\Tl}_{j_{l-1}, i_l}
        = \leftindex_{k}{T}^{(l-1)}_{\phi_{l-1}(1, j_{l-1}, i_l)}
        - \leftindex_{k}{T}^{(l-1)}_{\phi_{l-1}(-1, j_{l-1}, i_l)}.
    \]
    Define
    \[
        \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(1, j_{l-1}, i_l)}
        = \leftindex_{k}{\alpha}_{i_{l}} \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(1, j_{l-1}, i_l)}.
    \]
    The inductive assumption implies
    \( \leftindex_{k}{\hat{T}}^{(l-1)} \in \calC^{(l-1)} \).
    Moreover,
    \( \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(1, j_{l-1}, i_l)}
    \in \calK^{(l-1)}_{j_{l-1}} \) since \( \calK^{(l-1)}_{j_{l-1}} \) is a
    convex cone.
    Thus,
    \begin{equation*}
        \leftindex_{k}{\hat{T}}^{(l)}_{i_{l}} :=
        \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(1, j_{l-1}, i_l)}
        - \leftindex_{k}{\hat{T}}^{(l-1)}_{\phi_{l-1}(-1, j_{l-1}, i_l)}
    \end{equation*}
    must be in \( \calCl \).
    We conclude that \( \hat{T}^{(l)} \in \calCl \) since it is a non-negative
    combination of elements of \( \calCl \) and \( \calCl \) is a convex cone
    by \cref{lemma:convex-recursive-cone-constraint}.

\end{proof}

\constraintConvexification*
\begin{proof}
    Recall that
    \( \calG^{(1)} = \R^{d_0 \times d_1} = \calC^{(1)} \);
    this establishes our base case.

    Now, assume that
    \( \calG^{(l)} = \calC^{(l)} \) for some \( 1 \leq l < k  \).
    Let \( \Tll \in \calG^{(l+1)} \) be arbitrary.
    By membership in \( \calG^{(l+1)} \), \( \Tll \) satisfies
    \[
        \Tll_{j_1, \ldots, j_l}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \Tl_{j_1, \ldots, j_{l-1}, i_l}
        \otimes W_{i_l}^{(l+1)},
    \]
    where \( \Tl_{i_l} \in \calKl_{j_l} \)
    and \( \Wll \in \R^{d_l \times d_{l+1}} \).
    For each \( i_{l+1} \in [d_{l+1}] \), this implies
    \begin{align*}
        \Tll_{j_l, i_{l+1}}
         & = \sum_{i_l \in \calI_{j_l}^{(l)}} \Tl_{i_l}
        \cdot W_{i_l, i_{l+1}}^{(l+1)}                       \\
         & = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0)
        - \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} < 0).
    \end{align*}
    We can now define
    \[
        \hat{T}^{(l)}_{ \phi_l(1, j_l, i_{l+1})}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} \geq 0)
    \]
    and
    \[
        \hat{T}^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}
        = \sum_{i_l \in \calI_{j_l}^{(l)}} \rbr{\Tl_{i_l}
            \cdot \abs{W_{i_l, i_{l+1}}^{(l+1)}}} \mathds{1}(W_{i_l, i_{l+1}}^{(l+1)} < 0),
    \]
    to show that
    \( \Tll_{j_l, i_{l+1}}
    = \hat{T}^{(l)}_{ \phi_l(1, j_l, i_{l+1})} - \hat{T}^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}\).
    Moreover,
    \( \calKl_{j_l} \) is a convex cone, so
    \( \hat{T}^{(l)}_{ \phi_l(1, j_l, i_{l+1})}, \hat{T}^{(l)}_{ \phi_l(-1, j_l, i_{l+1})}
    \in \calKl_{j_l} \) as required.
    Finally, since \( \Tl \in \calG^{(l)} = \calC^{(l)} \), we can invoke
    \cref{lemma:cone-scaling} to show that \( \hat{T}^{(l)} \in \calC \).
    Note that these inclusions require \( d_{l} \geq 2 p_l d_{l+1} \),
    which holds by assumption.
    We conclude that \( \Tll \in \calCll \) and so
    \( \calG^{(l+1)} \subseteq \calC^{(l+1)} \).

    Now we show that the reverse inclusion is satisfied under
    the same assumption that \( d_{l} \geq 2 p_l d_{l+1} \).
    Towards that goal, take
    \( \Tll \in \calC^{(l+1)} \) to be arbitrary.
    By definition of \( \calC^{(l+1)} \), there exists
    \( \Tl \in \calCl \) such that
    \begin{align*}
        \Tll_{j_l, i_{l+1}}
         & = \Tl_{ \phi_l(1, j_l, i_{l+1})} - \Tl_{ \phi_l(-1, j_l, i_{l+1})},
    \end{align*}
    where \( \Tl_{ \phi_l(1, j_l, i_{l+1})}, \Tl_{ \phi_l(-1, j_l, i_{l+1})}
    \in \calK_{j_l} \).
    That is,
    \[
        \Tll_{j_1, \ldots, j_l}
        = \sum_{i_{l+1}=1}^{d_{l+1}}
        \begin{bmatrix}
            \Tl_{j_1, \ldots,  \phi_l(1, j_l, i_{l+1})} \otimes e_{i_{l+1}}
            - \Tl_{j_1, \ldots,  \phi_l(-1 j_l, i_{l+1})} \otimes e_{i_{l+1}}
        \end{bmatrix},
    \]
    where we are using standard basis vectors \( e_{i_{l+1}} \) to embed
    the tensors into the correct dimensions of \( \Tll \).
    Since \( C^{(l)} = G^{(l)} \),
    \( \Tl \in \calG^{(l)} \) as required.
    Choosing \( W^{(l+1)} \in \R^{d_{l} \times d_{l+1}} \) to be the block matrix
    with \( p_l + 1 \) blocks given by
    \[
        W^{(l+1)}
        =
        \begin{bmatrix}
            I      \\
            \vdots \\
            I      \\
            -I     \\
            \vdots
            -I     \\
            \mathbf{0}
        \end{bmatrix},
    \]
    we can now write
    \begin{align*}
        \Tll_{j_1, \ldots, j_{l}}
         & = \sum_{i_{l} \in \calI_{j_l}^{(l)}}
        \Tl_{j_1, \ldots, j_{l-1}, i_l}
        \otimes W_{i_{l}},
    \end{align*}
    where the activation indices are given by
    \[
        \calI_{j_l}^{(l)}
        = \cbr{ \phi_l(1, j_l, i_{l+1}) : i_{l+1} \in [d_{l+1}]}
        \cup
        \cbr{ \phi_l(2, j_l, i_{l+1}) : i_{l+1} \in [d_{l+1}]}.
    \]
    Thus, we need \( 2 d_{l+1} \) outer-products per \( j_l \in [p_l] \) to
    represent \( \Tll \).
    This is possible since \( d_l \geq 2 p_l d_{l+1} \),
    by assumption.
    We conclude that \( \Tll \in \calGll \) and thus \( \calGll = \calCll \)
    as required.
\end{proof}

\tensorSolutionMapping*
\begin{proof}
    The proof reverses the layer elimination lemma.
    For any \( l \in [k-1] \), we have
    \begin{align*}
        X^{(l+1)} \odot T^{(l+1)}
         & = \sum_{j_{l}=1}^{p_l} X^{(l+1)}_{j_{l}} \odot T^{(l+1)}_{j_{l}}          \\
         & = \sum_{j_{l}=1}^{p_l} \sum_{i_{l+1}=1}^{d_{l+1}}
        \rbr{X^{(l+1)}_{j_{l}} \odot T^{(l+1)}_{j_{l}, i_{l+1}}} \otimes e_{i_{l+1}} \\
         & = \sum_{j_{l}=1}^{p_l} \sum_{i_{l+1}=1}^{d_{l+1}}
        \rbr{X^{(l+1)}_{j_{l}} \odot
        \sbr{T^{(l)}_{\phi_l(1, j_{l}, i_{l+1})} - T^{(l)}_{\phi_l(-1, j_{l}, t i_{l+1})}}}
        \otimes e_{i_{l+1}}                                                          \\
         & = \sum_{i_{l}=1}^{d_{l}}
        D^{(l)}_{i_{l}} \rbr{X^{(l)} \odot
        T^{(l)}_{i_{l}}} \otimes W^{(l+1)}_{i_{l}}                                   \\
         & = \sum_{i_{l}=1}^{d_{l}}
        \rbr{X^{(l)} \odot T^{(l)}_{i_{l}}}_+ \otimes W^{(l+1)}_{i_{l}}              \\
         & = \rbr{X^{(l)} \odot T^{(l)}}_+ W^{(l+1)}.
    \end{align*}
    Applying this recursively from \( l = k - 1 \)
    and using \( X^{(1)} \odot T^{(1)} = X W^{(1)} \) completes the proof.
\end{proof}

\minimumSolutionWidth*
\begin{proof}
    Let \( \Tk \in \calCk \) be a solution to
    \cref{eq:convex-tensor-program}.
    We shall construct a solution with the desired width bounds from
    \( \Tk \).
    Let \( l \in [L-1] \) and suppose that the layer at \( l+1 \)
    has \( k \) non-zero outputs.
    That is, the set
    \[
        \calA(\Tll) = \cbr{i_{l+1} \in [d_{l+1}] : T^{(l+1)}_{i_{l+1}} \neq 0 },
    \]
    has \( k \) elements.
    Now, observe that for each \( i_{l+1} \in \calA(T^{(l+1)}) \),
    \begin{align*}
        X^{(l+1)} \odot T^{(l+1)}_{i_{l+1}}
         & = \sum_{j_{l} = 1}^{p_{l}}
        X^{(l+1)}_{j_l}  \odot T^{(l+1)}_{j_{l}, i_{l+1}} \\
         & = \sum_{j_{l} = 1}^{p_{l}}
        X^{(l+1)}_{j_l}  \odot
        \sbr{T^{(l)}_{\phi_{l}(1, j_{l}, i_{l+1})} - T^{(l)}_{\phi_{l}(-1, j_{l}, i_{l+1})}}.
    \end{align*}
    The non-zero terms in this sum must each be embedded into \( T^{(l)} \),
    meaning each contributes one to the width of the
    \( (l+1)^{\text{th}} \) layer.
    The set of non-zero terms is,
    \[
        \calQ_{i_{l+1}}(T^{(l)})
        = \cbr{q_{s, j_l, i_{l+1}} :=
            X^{(l+1)}_{j_l} \odot T^{(l)}_{\phi_{l}(1, j_{l}, i_{l+1})}
            : j_l \in [p_l], s \in \cbr{+1, -1},
            T^{(l)}_{\phi_{l}(s, j_{l}, i_{l+1})} \neq 0}
    \]
    Since each vector
    \( q_{s, j_l, i_{l+1}} \in \mathcal{N}_{i_{l+1}}(\Tl) \) is
    \( n \)-dimensional, \( \calQ_{i_{l+1}}(\Tl) \)
    contains a linearly dependent set if it contains are more than \( n \)
    vectors.
    Let
    \[
        \calN_{i_{l+1}}(T^{(l)})
        = \cbr{(s, j_l) : q_{s, j_l, i_{l+1}} \in \calQ_{i_{l+1}}(\Tl)}.
    \]
    Then linear dependence implies there exists
    \( \cbr{\alpha_(s,j_l) : (s,j_l) \in \calN_{i_{l+1}}(T^{(l)})} \)
    not all equal to zero such that
    \begin{align*}
        0 & = \sum_{(s,j_l) \in \calN_{i_{l+1}}(T^{(l)})} \alpha_{s, j_l} q_{s, j_l, i_{l+1}} \\
          & = \sum_{(s,j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        \alpha_{s,j_l} \rbr{X^{(l)}_{j_l} \odot T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}}          \\
          & = \sum_{(s,j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        \rbr{X^{(l+1)}_{j_l} \odot \sbr{\alpha_{s,j_l} T^{(l)}_{\phi_{l}(s,j_l,i_{l+1})}}}.
    \end{align*}
    Let \( (s^*, r^*) =
    \argmax_(s, j_l) \cbr{ \abs{\alpha_{s, j_l}} :
        (s, j_l) \in \calN_{i_{l+1}}(T^{(l)})} \)
    and define \( \tilde{\alpha}_{s, j_l} = \alpha_{s, j_l} / \alpha_{s^* ,r^*} \).
    Clearly \( \tilde{\alpha}_{s, j_l} \leq 1 \) for all \( (s, j_l) \) and
    \( \tilde{\alpha}_{(s^*, r^*)} = 1 \).
    Using this notation, we have
    \begin{align*}
        0 & = \sum_{(s, j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        \rbr{X^{(l)}_{j_l} \odot \sbr{\tilde{\alpha}_{s, j_l} T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}}}.
    \end{align*}
    This implies
    \begin{align*}
        X^{(l+1)} \odot T^{(l+1)}_{i_{l+1}}
         & = \sum_{(s, j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        X^{(l+1)}_{j_l} \odot
        \sbr{(1 + \tilde{\alpha}_{s, j_l}) T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}} \\
         & = \sum_{(s,j_l) \in \calN_{i_{l+1}}(T^{(l)})}
        X^{(l+1)}_{j_l} \odot
        \hat{T}^{(l)}_{\phi_{l}(s, j_l, i_{l+1})},
    \end{align*}
    where \( \hat{T^{(l)}} \) is the scaled version of \( T^{(l)} \)
    satisfying
    \( \hat{T}^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}
    = \sbr{(1 + \tilde{\alpha}_{s, j_l}) T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})}} \)
    for each \( (s, j_l) \in \calN_{i_{l+1}}(T^{(l)}) \) and
    \( \hat{T}^{(l)}_{\phi_{l}(s, j_l, i_{l+1})} = T^{(l)}_{\phi_{l}(s, j_l, i_{l+1})} \)
    otherwise.
    It is clear from the above calculation that \( \hat{T}^{(l)} \) preserves
    the predictions of \( T^{(l)} \)
    and maintains \( \hat{T}^{(l)}_{\phi_{l}(s, j_l, i_{l+1})} \in \calK_{j_l}^{(l)} \)
    since \( \tilde{\alpha}_{s, j_l} \geq 0 \) for each \( (s, j_l) \).
    Additionally, \( \hat{T}^{(l)} \in \calCl \) by \cref{lemma:cone-scaling}.
    Defining
    \[
        \hat{T}^{(l+1)}_{s, j_l, i_{l+1}} =
        \hat{T}^{(l)}_{\phi_{l}(1, j_l, i_{l+1})}
        - \hat{T}^{(l)}_{\phi_{l}(-1, j_l, i_{l+1})}
    \]
    we see that \( \hat{T}^{(l+1)} \in \calC^{(l+1)} \)
    and thus \( \hat{T}^{(l+1)} \) is also optimal.
    Moreover,
    \( \calA(\hat{T}^{(l)}) \leq \calA(\hat{T}^{(l)}) - 1 \).
    In other words, the width of \( m_l(\Tk) \) is reduced by one.

    For each \( i_{l+1} \), we can repeat this argument until the vectors
    in \( \calQ_{i_{l+1}}(\hat{T}^{(l)}) \) are linearly independent.
    This implies there exists optimal \( \hat{T}^{(l)} \)
    for which \( \abs{\calQ_{i_{l+1}}(\hat{T}^{(l)})} \leq n \).
    Repeating this procedure independently for each \( i_{l+1} \)
    for which \( T^{(l+1)}_{i_{l+1}} \neq 0 \), we construct
    \( \hat{T}^{(l+1)} \) with
    \( m_{l}(\hat{T}^{(l)}) \leq k n \).
    Noting that \( k \leq c \) (the number of network outputs) when
    \( l = k - 1 \) and applying a simple inductive argument completes the
    proof.
\end{proof}

\deepConvexReformulation*
\begin{proof}
    Let \( \Tk \in \calCk \) be the solution to \cref{eq:convex-tensor-program}
    which is guaranteed to exist by \cref{prop:convex-minimal-solution-width}.
    This solution is sparse and satisfies \( m_l = \calA\rbr{\Tl} \leq c n^{L - l} \).
    \cref{lemma:convex-tensor-solution-mapping} now implies that there
    exists a optimal sparse neural network \( \theta^* \) with at
    most \( m_l \) neurons in the \( l^{\text{th}} \) layer.
    Removing zero neurons results in the architecture
    \( \Delta' = \rbr{m_1, \ldots, m_{L-1}} \)
    with associated parameter space \( \Theta_{\Delta'} \).
    Let \( \Delta \) be any architecture satisfying
    \( d_l \geq 2 p_l d_{l+1} \).
    Now can obtain
    \begin{align*}
        \min_{\Tk \in \calCk} \calL(\Xk \odot \Tk, y)
         & = \min_{\theta \in \Theta} \calL(f_\theta(X), y)              \\
         & \leq \min_{\theta \in \Theta_{\Delta'}} \calL(f_\theta(X), y) \\
         & \leq \calL(f_{\theta^*}(X), y)                                \\
         & = \min_{\Tk \in \calCk} \calL(\Xk \odot \Tk, y),
    \end{align*}
    which shows that the two problems have the same optimal value.
\end{proof}

\subsection{Deep Gated ReLU Networks: Proofs}

\gatedReLUActivationSet*
\begin{proof}
    The result is obtained by induction.
    Since \( \calD^{(1)}_{\text{gated}} = \calD^{(1)} \), every network
    \( g_{\theta_{1}} \) can be embedded into \( g^{\text{max}}_{\theta_{1}} \),
    as required.

    Now assume that the desired result holds for some \( l \geq 1 \).
    Since \( g_{\theta_{l}} \) can be embedded into \( g^{\text{max}}_{\theta_{l}'} \)
    for some \( \theta_l' \),
    any pattern generated as
    \[
        \tilde{D}^{(l+1)}_{i_{j+1}}
        = \emph{\diag}\rbr{\mathds{1} \rbr{g_{\theta_{l}}(X) u \geq 0}},
    \]
    can be generated by the corresponding maximal gated ReLU network:
    \[
        \tilde{D}^{(l+1)}_{i_{j+1}}
        = \emph{\diag}\rbr{\mathds{1} \rbr{g^{\text{max}}_{\theta_{l}'}(X) u \geq 0}}.
    \]
    That is,
    \( \tilde{D}^{(l+1)}_{i_{j+1}} \in \calD^{(l+1)} \)
    as required.
    This also implies that we can represent any \( g_{\theta_{l+1}} \)
    as some maximal network \( g^{\text{max}}_{\theta_{l+1}'} \),
    which completes the proof.
\end{proof}

\gatedReLULayerElimination*
\begin{proof}
    We start by defining a suitable \( \Tll \) as follows:
    \begin{equation}\label{eq:gated-t-plus-characterization}
        \Tll_{j_1, \ldots, j_l}
        = \Tl_{j_1, \ldots, j_{l-1}, j_l} \otimes W^{(l+1)}_{j_l}.
    \end{equation}
    Using this notation, we re-write the sum
    in \cref{eq:gated-layer-to-tensorize} as follows:
    \begin{align*}
        Z^{(l+2)}
         & = \mathbf{D}^{(l+1)}_\text{gated} \cdot
        \sbr{\sum_{j_l=1}^{\tilde{p}_l} \rbr{\Dl_{j_l} \sbr{\Xl \odot \Tl_{j_l}}}
        \otimes W^{(l+1)}_{j_l}}                                              \\
         & = \mathbf{D}^{(l+1)}_\text{gated} \cdot
        \sbr{\sum_{j_l=1}^{\tilde{p}_l} \rbr{\Dl_{j_l} \Xl \odot \sbr{\Tl_{j_l}
        \otimes W^{(l+1)}_{j_l}}}}                                            \\
         & = \mathbf{D}^{(l+1)}_\text{gated} \cdot
        \sbr{\sum_{j_l=1}^{\tilde{p}_l} \rbr{\Dl_{j_l} \Xl \odot \Tll_{j_l}}} \\
         & = \mathbf{D}^{(l+1)}_\text{gated} \cdot \sbr{\Xll \odot \Tll}.
    \end{align*}
    It should be clear from the way that we constructed \( \Tll \) that
    it is contained \( \calH^{(l+1)} \).
    This establishes one direction of theorem.
    The other direction can be shown by instead assuming that
    \( \Tll \in \calH^{(l+1)} \) and then noting that the characterization
    in \cref{eq:gated-t-plus-characterization} immediately follows for some
    \( \Tl \in \calH^{(l)} \) and \( W^{(l+1)} \in \R^{p_l \times p_{l+1}} \).
    The preceding calculation is then reversed to obtain
    \cref{eq:gated-layer-to-tensorize}.
\end{proof}

%\convexGatedReLU*
\begin{proof}
    Let \( \Tk \) be a solution to \cref{eq:convex-gated-relu-program}.
    Since \( \Tk \in \R^{d_0 \times \tilde{p}_1 \times \cdots \times \tilde{p}_{L-1} \times c} \),
    it admits a canonical polyadic decomposition as
    \[
        \Tk = \sum_{r=1}^{k} a^{(0)}_{r} \otimes \cdots \otimes a^{(L)}_{r},
    \]
    where \( r \) is upper-bounded by \( d_0 \tilde{p}_1 \cdots \tilde{p}_{L-1} c  \).
    In particular, we can write
    \[
        \Tk \sum_{r=1}^{k} a^{(0)}_{r} \otimes \cdots \otimes a^{(L)}_{r}
    \]

    The prediction function is
    \begin{align*}
        \tilde{X}^{(L)} \odot \Tk
         & = \sum_{j_1=1}^{\tilde{p_1}} \cdots \sum_{j_{L-1}=1}^{\tilde{p_{L-1}}}
        \tilde{X}^{(L)}_{j_1, \ldots, j_{L-1}} \Tk                                \\
         & = \sum_{j_1=1}^{\tilde{p_1}} \cdots \sum_{j_{L-1}=1}^{\tilde{p_{L-1}}}
        \tilde{X}^{(L)}_{j_1, \ldots, j_{L-1}} \Tk
    \end{align*}
\end{proof}
