Now we analyze the structure of \( \calGl \) with the goal of establishing a
fully convex reformulation for deep ReLU networks.
This recursive constraint set encodes several key features of ReLU networks:
(i) weight sharing, through \( \Wll \), (ii) low-rank neurons, through the
outer-products \( \Tl_{j_1, \ldots, j_{l-1}, i_l} \otimes \Wll_{j_l} \),
(iii) the layer width, through
\( \sum_{j_l=1}^{p_l} \abs{\calI_{j_l}^{(l)}} \leq d_l \),
and (iv) activation cone membership.
Out of these constraints, (i) to (iii) are affected by the width of the
network.
We will show that these constraints vanish for sufficiently wide models.

Let \( \phi_l : \cbr{1, -1} \times [p_l] \times [d_{l+1}] \rightarrow [2 p_l \cdot d_{l+1}] \)
be a one-to-one function from the index triplets
\( (s, j_l, i_{l+1}) \) to a unique index in \( [2 p_l \cdot d_{l+1}] \).
Using this notation, consider the tensor constraint defined recursively by
\( \calC^{(1)} = \R^{d_0 \times d_1} \) and
\begin{equation}\label{eq:recursive-cone-constraint}
    \begin{aligned}
        \calC^{(l+1)}
         & = \bigg\{
        \Tll \in \R^{d_0 \times p_1 \cdots \times p_{l} \times d_{l+1}}
        :
        \Tll_{j_{l}, i_{l+1}}
        = \Tl_{ \phi_l(1, j_l, i_{l+1})}
        - \Tl_{ \phi_l(-1, j_l, i_{l+1})},
        \text{ where}   \\
         & \hspace{3em}
        \Tl_{\phi_l(1, j_l, i_{l+1})}, \Tl_{ \phi_l(-1, j_l, i_{l+1})}
        \in \calK^{(l)}_{j_{l}}, \text{ and }
        \Tl = \calC^{(l)}
        \bigg\}
    \end{aligned}
\end{equation}
The construction of \( \calCl \) requires the dimension of the
last order, \( d_{l} \), to satisfy \( d_{l} \geq 2 p_{l} d_{l+1} \);
otherwise, we could not embed \( \Tll \) into \( \Tl \).
It is straightforward to show that \( \calC^{(l)} \) is a convex cone
(\cref{lemma:convex-recursive-cone-constraint}).
More importantly, \( \calC^{(l)} \) is exactly the constraint set obtained
by making the network wide enough that the weight sharing, low-rank,
and width constraints in \( \calGl \) vanish.

\begin{restatable}[Constraint Convexification]{proposition}{constraintConvexification}%
    \label{prop:constraint-convexification}
    If \( d_l \geq 2 p_l d_{l+1} \) for every \( l \in [L-1]  \), then
    \( \calG^{(L)} = \calC^{(L)} \).
\end{restatable}

Although short, \cref{prop:constraint-convexification} is the critical
proposition for our convex reformulation.
It says that if the network is sufficiently wide, \( \calGk \) is convex and
Problem~\ref{eq:non-convex-tensor-program} is a fully convex optimization
program.
Recalling that the \( d_L = c \), the condition \( d_l \geq 2 p_l d_{l+1} \)
is equivalent to \( d_l \geq c 2^{L-l} \prod_{i = l+1}^{L} p_l \).
In other words, the network must be shaped like a pyramide, where each
layer is multiplicatively larger than the succeeding one,
for the constraint set to convexify and \( \calGk = \calCk \) to hold.
This leads to the following corollary.
\begin{corollary}\label{cor:convex-tensor-program}
    If \( d_l \geq c \, 2^{L-l} \prod_{i = l}^{L-1} p_l \) for every
    \( l \in [L-1]  \), then
    \begin{equation*}
        \min_{\theta = (W^{(1)}, \ldots, \Wk)} \calL\rbr{f_{\theta}(X), y},
    \end{equation*}
    is equivalent to the convex tensor program
    \begin{equation}\label{eq:convex-tensor-program}
        \min_{\Tk}
        \calL\rbr{\Xk \odot \Tk, y} \quad \text{s.t.} \quad
        \Tk \in \calC^{(k)}.
    \end{equation}
\end{corollary}

Next we show that the solution mapping for \cref{eq:convex-tensor-program}
relies on the recursive decomposition specified by membership in \( C^{(k)} \).
Moreover, the number of neurons in each layer corresponds to the number of
non-zero activation patterns.

\begin{restatable}[Solution Mapping]{lemma}{tensorSolutionMapping}%
    \label{lemma:convex-tensor-solution-mapping}
    Assume \( d_l \geq 2 p_l d_{l+1} \) for all \( l < k \).
    Fix \( \Tk \in \calCk \) and for \( 1 < l < k \), let \( \Tl \in \calCl \)
    be the associated recursive factor.
    Define the set of non-zero neurons as
    \[
        \calA(\Tl) = \cbr{i_{l} \in [d_l] : \Tl_{i_{l}} \neq 0}.
    \]
    For every \(s \in \cbr{1, -1} \), \( j_l \in [p_l] \), and
    \( i_{l+1} \in [d_{l+1}] \), the associated rows of \( W^{(l+1)} \) are
    given by
    \[
        W^{(l+1)}_{\phi_l(s, j_l, i_{l+1})}
        = s \cdot e_{i_{l+1}} \mathds{1}\rbr{\phi_l(s, j_l, i_{l+1}) \in \calA(\Tl)}.
    \]
    Finally, let \( W^{(1)} = T^{(1)} \) and
    \( \theta = \rbr{W^{(1)}, \ldots, W^{(k)}} \).
    Then \( X^{(k)} \odot T^{(k)} = f_\theta(X) \).
\end{restatable}

Given any tensor model \( \Tk \in \calCk \), \cref{lemma:convex-tensor-solution-mapping}
provides a simple way of computing an equivalent neural network \( f_\theta \).
Moreover, it shows that the number of non-zero neurons in the
\( l^{\text{th}} \) layer of the network is given by
\( m_{l}(\Tk) = \abs{\calA(\Tl)} \).
Collecting these widths as \( \Delta(\Tk) = \rbr{m_1(\Tk), \ldots, m_{L-1}(\Tk)} \),
the optimal solutions to Problem~\ref{eq:convex-tensor-program} define
a Pareto frontier of minimal-width networks with the same optimal
value as the convex reformulation.
Let
\begin{equation}\label{eq:tensor-optima}
    \calO^* = \argmin_{\Tk \in \calCk} \calL(\Xk \odot \Tk, y)
\end{equation}
be the set of optimal solutions to Problem~\ref{eq:convex-tensor-program}.
Then the frontier of minimal widths is given as follows:
\begin{equation}
    \calD^* :=
    \cbr{\Delta(\Tkopt)
        : \nexists T^{(L)'} \in \calO^*,
        \Delta(T^{(L)'}) \prec \Delta(\Tkopt)}.
\end{equation}

We now exhibit a solution \( \Tkopt \in \calO^* \) which allows us to
upper bound \( m^*_l \) for at least one point
\( \Delta^* = (m^*_1, \ldots, m^*_{L-1}) \in \calD^* \).

\begin{restatable}{proposition}{minimumSolutionWidth}\label{prop:convex-minimal-solution-width}
    There exists at least one solution to the convex tensor optimization
    program in \cref{eq:convex-tensor-program} satisfying
    \( m_l(\Tk) \leq c n^{L - l} \) for every
    \( l \in \cbr{1, \ldots, L-1} \).
\end{restatable}

As an immediate result, we obtain the main theorem of this section.

\begin{restatable}{theorem}{deepConvexReformulation}\label{thm:deep-convex-reformulation}
    If for every \( l \in [L-1]  \) it holds that
    \( d_l \geq m^{*}_{l} \) for some threshold \( m^{*}_{l} \leq c \, n^{L - l} \),
    then
    \begin{equation*}
        \min_{\theta = (W^{(1)}, \ldots, \Wk)} \calL\rbr{f_{\theta}(X), y},
    \end{equation*}
    is equivalent to the convex tensor program
    \begin{equation}
        \min_{\Tk}
        \calL\rbr{\Xk \odot \Tk, y} \quad \text{s.t.} \quad
        \Tk \in \calC^{(k)}.
    \end{equation}
\end{restatable}

In comparison to \cref{cor:convex-tensor-program}, this theorem reduces
the width requirement on the \( l^{\text{th}} \) layer of the neural network
from \( d_l \geq c \, 2^{(L- l)} \prod_{i=l}^{L-1} p_l \) to
\( d_l \geq m^*_{l} \) where \( m^*_{l} \) is an unknown threshold
bounded as \( m^*_{l} \leq c \, n^{L - 1} \).
If the number of layers in the network, \( L \), is fixed ahead of time, then
this is fully polynomial in the input data.
Moreover, the bound on \( m^*_l \) is tight with existing results on
convex reformulations.
For example, \citep{sahiner2021vector} show that a convex reformulation exists
for two-layer ReLU networks with vector outputs if \( d_1 \geq m_1^* \) for
\( m_1^* \leq c \, n \).
Specializing \cref{thm:deep-convex-reformulation} to \( L = 2 \) yields exactly
the same condition.

\subsection{Deep Gated ReLU Networks}\label{sec:deep-gated-relu}

In \cref{chapter:fast-optimization}, we showed that two-layer neural networks
with Gated ReLU activations admit a simple, unconstrained convex
reformulation.
Given a fixed dataset \( (X, y) \), we also showed that gated ReLU networks
form the same hypothesis space as two-layer ReLU networks --- that is, as long
as no singular activation patterns are included in the Gated ReLU network.
This allowed us to develop fast, unconstrained optimization methods
which either exactly solve (in the unregularized setting) or approximate
(in the regularized setting) the two-layer ReLU problem.
Now we consider deep gated ReLU networks and develop an equivalent theory.

Recall that the gated ReLU activation function is given by
\[
    \phi_G\rbr{Z^{(l)}, W^{(l)}}
    = \mathds{1}\rbr{Z^{(l)} G^{(l)} \geq 0} \cdot (Z^{(l)} W^{(l)}),
\]
where \( G \in \R^{d_{(l-1)} \times d_{l}} \) is a matrix of the gate vectors,
the indicator function is taken element-wise over \( Z^{(l)} G^{(l)} \),
and \( \cdot \) indicates the element-wise product.
A deep gated ReLU network is then defined by
\begin{equation}\label{eq:deep-gated-relu}
    g_\theta(X) = \Zk \Wk,
    \quad \text{where} \quad
    \Zll = \phi_G\rbr{\Zl, G^{(l)}}.
\end{equation}
While the activations patterns generated by
\( \mathds{1}\rbr{Z^{(1)} G^{(1)} \geq 0} \) in the first-layer of a gated
ReLU network are necessarily the same as those generated by a ReLU network, the
space of attainable features \( Z^{(2)} \) is not equivalent unless all
singular activation patterns are omitted (\cref{thm:cone-elimination}).
We handle this issue by defining a maximal deep
gated ReLU network with explicit activation patterns.

For \( L = 1 \), the maximal gated ReLU network is provided by
\cref{chapter:fast-optimization} and has the form
\begin{equation}\label{eq:maximal-gated-relu}
    g^{\text{max}}_{\theta_{1}}(X)
    = \begin{bmatrix}
        D^{(1)}_{1} X W^{(1)}_{\cdot, 1} \cdots
        D^{(1)}_{p_1} X W^{(1)}_{\cdot, p_1}
    \end{bmatrix}
    = \mathbf{D}^{(1)} \cdot X W^{(1)},
\end{equation}
where \( D^{(1)}_{j_1} \in \calD^{(1)} \) and
\( \mathbf{D}^{(1)} \) is the matrix whose \( i_1^{\text{th}} \)
column is given by the diagonal of \( D^{(1)}_{i_l} \).
Since \( \calD^{(1)} \) has no dependence on the network structure, this
captures the first-layer feature map of any two-layer Gated ReLU network.
Going forward, we use \( \calD^{(l)}_{\text{gated}} \) to denote the
set of all activation patterns achievable by an \( l \)-layer gated ReLU network,
where we have already seen that
\( \calD^{(1)}_{\text{gated}} = \calD^{(1)} \).
The following definition uses this as a base case to define the maximal gated
ReLU network and its associated activation patterns for \( l > 1 \).
\begin{definition}
    The gated ReLU activation patterns at layer \( l+1 \) are given by
    \begin{equation}
        \calD^{(l+1)}_{\text{gated}} =
        \cbr{
        \tilde{D}^{(l+1)}_{j_{l+1}} =
        \emph{\diag}\rbr{\mathds{1}\rbr{g^{\text{max}}_{\theta_{l}}(X) u \geq 0}}
        : \theta^{(l)} \in \Theta_\Delta^{(l)}, u \in \R^{d_{l+1}}
        },
    \end{equation}
    where \( g^{\text{max}}_{\theta_{l}} \) is the maximal gated ReLU
    network at layer \( l \) and is defined to be
    \begin{equation}
        g^{\text{max}}_{\theta_{l}}(X)
        = \begin{bmatrix}
            \tilde{D}^{(l)}_{1} g^{\text{max}}_{\theta_{l-1}}(X) W^{(l)}_{\cdot, 1}  \cdots
            \tilde{D}^{(l)}_{p_1} g^{\text{max}}_{\theta_{l-1}}(X) W^{(l)}_{\cdot, \tilde{p}_l}
        \end{bmatrix}
        = \mathbf{D}_{\text{gated}}^{(l)} \cdot g^{\text{max}}_{\theta_{l-1}}(X) W^{(l)}.
    \end{equation}
\end{definition}

Note that we use \( \tilde{p}_l = \abs{\calD^{(l)}_{\text{gated}}} \) and
\( \tilde{D}^{(l)}_{j_l} \in \calD^{(l)}_{\text{gated}} \) to denote
the number of gated ReLU activation patterns and the specific patterns,
respectively.
The next lemma shows that this characterization captures all possible
gated ReLU models.
\begin{restatable}{lemma}{gatedReLUActivationSet}%
    \label{lemma:gated-relu-activation-patterns}
    Fix any architecture \( \Delta \).
    The set \( \calD^{(l)}_{\text{gated}} \) contains all activation
    patterns that can be generated by a geted ReLU network parameterized 
    by \( \theta_{l-1} \in \Theta_{\Delta^{l-1}} \).
    That is, for any choice of \( \theta_{l-1} \) and \( u \),
    \[
        \emph{\diag}\rbr{\mathds{1} \rbr{g_{\theta_{l-1}} u \geq 0}}
        \in \calD^{(l)}_{\text{gated}}.
    \]
    As a result, any gated ReLU network can written as a maximal
    gated ReLU network.
\end{restatable}
